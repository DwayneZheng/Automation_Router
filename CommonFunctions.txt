    waitElementDisplayed    ${Settings-Internet-IPv4-L2TP-My Internet Connection is-Select}    3    true
    click    ${Settings-Internet-IPv4-L2TP-My Internet Connection is-Select}
    waitElementDisplayed    ${Settings-Internet-IPv4-L2TP-L2TP-SelectChild}    3    true
    click    ${Settings-Internet-IPv4-L2TP-L2TP-SelectChild}
    waitElementDisplayed    ${Settings-Internet-IPv4-L2TP-L2TP Server-Input}    10    true
    clearText    ${Settings-Internet-IPv4-L2TP-L2TP Server-Input}
    sendtext    ${Settings-Internet-IPv4-L2TP-L2TP Server-Input}    192.168.11.254    #L2TP Server Input
    waitElementDisplayed    ${Settings-Internet-IPv4-L2TP-Username-Input}    10    true
    clearText    ${Settings-Internet-IPv4-L2TP-Username-Input}
    sendtext    ${Settings-Internet-IPv4-L2TP-Username-Input}    dlink    #Username Input
    waitElementDisplayed    ${Settings-Internet-IPv4-L2TP-Password-Input}    10    true
    clearText    ${Settings-Internet-IPv4-L2TP-Password-Input}
    sendtext    ${Settings-Internet-IPv4-L2TP-Password-Input}    dlink    #Password Input
    sleep    1s
